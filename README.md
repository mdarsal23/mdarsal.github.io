    // Move the snake in the current direction
    int nextX = headX;
    int nextY = headY;
    switch (currentDirection) {
        case UP: nextY--; break;
        case DOWN: nextY++; break;
        case LEFT: nextX--; break;
        case RIGHT: nextX++; break;
    }

        // Turn 90 degrees clockwise
        switch (currentDirection) {
            case UP: changeDirection(RIGHT); break;

    headX = nextX;
    headY = nextY;
